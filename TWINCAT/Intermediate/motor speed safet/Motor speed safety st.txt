rpm := ((analog_current_input - 4.0) / 16.0) * 5000.0;

IF rpm < 1000 THEN
    MotorSpeed := 1000;
ELSIF rpm > 5000 THEN
   MotorSpeed := 5000;
ELSIF rpm >= 1000 AND rpm <= 5000 THEN
    MotorSpeed := rpm;
END_IF
motorspeed_uint := REAL_TO_UINT(rpm);
